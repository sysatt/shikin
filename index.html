// ============================================================ // ATTEND.SYS — WITH MENTOR PHONE LOOKUP // + EDIT SESSION + KELAS GANTI + SPLIT SUPPORT // ============================================================ var SPREADSHEET_ID = "1x3E6812SbE_i9bArkxZvqajdFKbwN1JgkBgN8dDeU-M"; function doGet(e) { var params = e ? e.parameter : {}; if (params.action === "photo") return getPhoto(params.id, params.fileId); if (params.action === "savebulk") return saveBulkGet(params); if (params.action === "students") return getStudents(); return getStudents(); } function getStudents(){ var ss = SpreadsheetApp.openById(SPREADSHEET_ID); var mentorPhoneMap = {}; try { var fonSheet = ss.getSheetByName("FON NUMBER"); if(fonSheet){ var fonData = fonSheet.getDataRange().getValues(); for(var i = 3; i < fonData.length; i++){ var namStaf = String(fonData[i][2] || "").trim().toUpperCase(); var noTel = cleanPhone(fonData[i][5]); if(namStaf && noTel) mentorPhoneMap[namStaf] = noTel; } } } catch(e) {} function lookupMentorPhone(mentorName){ if(!mentorName) return ""; var upper = mentorName.trim().toUpperCase(); if(mentorPhoneMap[upper]) return mentorPhoneMap[upper]; var bestMatch = ""; var bestScore = 0; var keys = Object.keys(mentorPhoneMap); for(var k = 0; k < keys.length; k++){ var key = keys[k]; if(key.includes(upper) || upper.includes(key)){ var score = Math.min(key.length, upper.length); if(score > bestScore){ bestScore = score; bestMatch = key; } continue; } var keyWords = key.split(/\s+/); var inputWords = upper.split(/\s+/); var matches = 0; inputWords.forEach(function(w){ if(w.length > 2 && keyWords.indexOf(w) >= 0) matches++; }); var threshold = Math.max(2, Math.ceil(inputWords.length * 0.5)); if(matches >= threshold){ var score = matches * 10; if(score > bestScore){ bestScore = score; bestMatch = key; } } } return bestMatch ? mentorPhoneMap[bestMatch] : ""; } var sheet = ss.getSheets()[0]; var data = sheet.getDataRange().getValues(); var headers = data[0]; var normalizedHeaders = headers.map(function(h){ return normalize(h); }); function col(keyword){ keyword = normalize(keyword); for (var i = 0; i < normalizedHeaders.length; i++) { if (normalizedHeaders[i].includes(keyword)) return i; } return -1; } var colMap = { kelas: col("registered class"), name: col("nama penuh"), nickname: col("nama panggilan"), id: col("nombor id"), phone: col("no telefon"), mentor: col("mentor"), guardian: col("nama penjaga"), guardianPhone: col("telefon ibu"), resident: col("status kediaman"), photo: col("gambar") }; var rows = []; for (var i = 1; i < data.length; i++) { var row = data[i]; if (colMap.id === -1 || !row[colMap.id]) continue; var kelasRaw = String(row[colMap.kelas] || "").toUpperCase().replace(/\s+/g, ""); var kelas = "DCD2A"; if (kelasRaw.includes("DCS3A")) kelas = "DCS3A"; else if (kelasRaw.includes("DCS3B")) kelas = "DCS3B"; else if (kelasRaw.includes("DIA") && kelasRaw.includes("3C")) kelas = "DIASAP3C"; var mentorName = String(row[colMap.mentor] || "").trim(); var mentorPhone = lookupMentorPhone(mentorName); rows.push({ id: String(row[colMap.id] || "").trim().toUpperCase(), name: String(row[colMap.name] || "").toUpperCase(), nickname: String(row[colMap.nickname] || "").replace(/\(.*?\)/g,"").trim(), phone: cleanPhone(row[colMap.phone]), mentor: mentorName, mentorPhone: mentorPhone, guardian: String(row[colMap.guardian] || "").trim(), guardianPhone: cleanPhone(row[colMap.guardianPhone]), resident: String(row[colMap.resident] || "").toLowerCase().includes("non") ? "non-resident" : "resident", kelas: kelas, photoFileId: extractDriveId(row[colMap.photo] || "") }); } return output({ success: true, students: rows }); } function saveBulkGet(params){ try { var records = JSON.parse(decodeURIComponent(params.records || "[]")); var sh = getAttSheet(); var dur = params.dur ? decodeURIComponent(params.dur) : ""; var tS = params.tS ? decodeURIComponent(params.tS) : ""; var tE = params.tE ? decodeURIComponent(params.tE) : ""; var sessType = params.sessType ? decodeURIComponent(params.sessType) : ""; var isEdit = params.edit === "true"; var gantiDate = params.gantiDate ? decodeURIComponent(params.gantiDate) : ""; var gantiTS = params.gantiTS ? decodeURIComponent(params.gantiTS) : ""; var gantiTE = params.gantiTE ? decodeURIComponent(params.gantiTE) : ""; if(!records.length) return output({ success: true, saved: 0 }); if(isEdit && tS && records.length > 0){ var targetClass = records[0].class; var targetDate = records[0].date; var allData = sh.getDataRange().getValues(); var updateMap = {}; records.forEach(function(r){ updateMap[r.id] = { status: r.status, time: r.time }; }); var headerRowIdx = -1; for(var i = 0; i < allData.length; i++){ var cellA = String(allData[i][0] || ''); if(cellA.includes(targetDate) && cellA.includes(tS)){ headerRowIdx = i; break; } } var updatedCount = 0; if(headerRowIdx >= 0){ for(var i = headerRowIdx + 1; i < allData.length; i++){ var rowDate = String(allData[i][0] || '').trim(); var rowClass = String(allData[i][1] || '').trim(); var rowId = String(allData[i][2] || '').trim().toUpperCase(); if(rowDate && rowDate !== targetDate) break; if(rowClass && rowClass !== targetClass) break; if(!rowId) continue; if(updateMap[rowId]){ var sheetRow = i + 1; var newStatus = updateMap[rowId].status; var newTime = updateMap[rowId].time; sh.getRange(sheetRow, 5).setValue(newStatus); sh.getRange(sheetRow, 6).setValue(newTime); var range = sh.getRange(sheetRow, 1, 1, 6); if(newStatus === "HADIR"){ range.setBackground("#c6efce"); sh.getRange(sheetRow, 5).setFontColor("#276221").setFontWeight("bold"); } else if(newStatus === "TIDAK HADIR"){ range.setBackground("#ffc7ce"); sh.getRange(sheetRow, 5).setFontColor("#9c0006").setFontWeight("bold"); } else { range.setBackground("#f2f2f2"); sh.getRange(sheetRow, 5).setFontColor("#999999").setFontWeight("normal"); } updatedCount++; } } var newLabel = buildHeaderLabel(targetDate, targetClass, tS, tE, dur, sessType, gantiDate, gantiTS, gantiTE); sh.getRange(headerRowIdx + 1, 1).setValue(newLabel); return output({ success: true, saved: updatedCount, mode: 'updated' }); } } var rowsToWrite = []; var formatsToApply = []; if(tS){ var lastRow = sh.getLastRow(); if(lastRow > 1) rowsToWrite.push(["", "", "", "", "", ""]); var label = buildHeaderLabel(records[0].date, records[0].class, tS, tE, dur, sessType, gantiDate, gantiTS, gantiTE); rowsToWrite.push([label, "", "", "", "", ""]); formatsToApply.push({ rowOffset: rowsToWrite.length - 1, type: "header" }); } records.forEach(function(r){ rowsToWrite.push([r.date, r.class, r.id, r.name, r.status, r.time]); formatsToApply.push({ rowOffset: rowsToWrite.length - 1, type: r.status }); }); var startRow = sh.getLastRow() + 1; sh.getRange(startRow, 1, rowsToWrite.length, 6).setValues(rowsToWrite); sh.getRange(startRow, 1, rowsToWrite.length, 1).setNumberFormat("@"); formatsToApply.forEach(function(f){ var absRow = startRow + f.rowOffset; var range = sh.getRange(absRow, 1, 1, 6); if(f.type === "header"){ range.merge().setBackground("#2d3250").setFontColor("#ffffff").setFontWeight("bold").setFontSize(10).setHorizontalAlignment("left").setVerticalAlignment("middle"); sh.setRowHeight(absRow, 26); if(sessType === "ganti"){ range.setBackground("#3d2800").setFontColor("#e8952f"); } else if(sessType === "tambahan"){ range.setBackground("#0d2b2b").setFontColor("#2ec4b6"); } else if(sessType === "split"){ range.setBackground("#1a2040").setFontColor("#c9a84c"); } } else if(f.type === "HADIR"){ range.setBackground("#c6efce"); sh.getRange(absRow, 5).setFontColor("#276221").setFontWeight("bold"); } else if(f.type === "TIDAK HADIR"){ range.setBackground("#ffc7ce"); sh.getRange(absRow, 5).setFontColor("#9c0006").setFontWeight("bold"); } else { range.setBackground("#f2f2f2"); sh.getRange(absRow, 5).setFontColor("#999999"); } }); return output({ success: true, saved: records.length, mode: 'appended' }); } catch(err) { return output({ success: false, error: err.toString() }); } } function buildHeaderLabel(date, kelas, tS, tE, dur, sessType, gantiDate, gantiTS, gantiTE){ var label = date + " " + kelas + " | MASA: " + tS + " – " + tE + " | TEMPOH: " + dur; if(sessType === "ganti"){ label += " | 🔁 KELAS GANTI"; if(gantiDate) label += " | Ganti untuk: " + gantiDate; if(gantiTS && gantiTE) label += " | Masa asal: " + gantiTS + " – " + gantiTE; } else if(sessType === "tambahan"){ label += " | ➕ KELAS TAMBAHAN"; } else if(sessType === "split"){ label += " | ⏱ SPLIT"; } return label; } function getAttSheet(){ var ss = SpreadsheetApp.openById(SPREADSHEET_ID); var sh = ss.getSheetByName("REKOD KEHADIRAN"); if(!sh){ sh = ss.insertSheet("REKOD KEHADIRAN"); sh.appendRow(["TARIKH","KEHADIRAN KELAS","NO ID","NAMA","STATUS","MASA"]); } return sh; } function getPhoto(studentId, fileId) { if (!fileId) return output({ success:false }); try { var file = DriveApp.getFileById(fileId); var blob = file.getBlob(); var b64 = "data:image/jpeg;base64," + Utilities.base64Encode(blob.getBytes()); return output({ success:true, id:studentId, photo:b64 }); } catch(err) { return output({ success:false }); } } function output(obj){ return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } function extractDriveId(url) { if (!url) return ""; var m = String(url).match(/[-\w]{25,}/); return m ? m[0] : ""; } function normalize(str){ return String(str).toLowerCase().replace(/\s+/g,"").replace(/[^\w]/g,""); } function cleanPhone(p){ if (!p) return ""; p = String(p).replace(/\D/g,""); if (p.startsWith("0")) p = "6" + p; return p; }

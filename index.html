// ============================================================ // ATTEND.SYS — FINAL CLOUD VERSION // ============================================================ function doGet(e) { var params = e ? e.parameter : {}; if (params.action === "photo") { return getPhoto(params.id, params.fileId); } if (params.action === "students") { return getStudents(); } return output({success:false}); } function getStudents(){ var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = sheet.getDataRange().getValues(); var headers = data[0]; var normalizedHeaders = headers.map(h => normalize(h)); function col(keyword){ keyword = normalize(keyword); for (var i = 0; i < normalizedHeaders.length; i++) { if (normalizedHeaders[i].includes(keyword)) return i; } return -1; } var colMap = { kelas: col("registered class"), name: col("nama penuh"), nickname: col("nama panggilan"), id: col("nombor id"), phone: col("no telefon"), mentor: col("mentor"), guardian: col("nama penjaga"), guardianPhone: col("telefon ibu"), resident: col("status kediaman"), photo: col("gambar") }; var rows = []; for (var i = 1; i < data.length; i++) { var row = data[i]; if (colMap.id === -1 || !row[colMap.id]) continue; var kelasRaw = String(row[colMap.kelas] || "").toUpperCase(); var kelas = "DCD2A"; if (kelasRaw.includes("DCS") && kelasRaw.includes("3A")) kelas = "DCS3A"; else if (kelasRaw.includes("DCS") && kelasRaw.includes("3B")) kelas = "DCS3B"; else if (kelasRaw.includes("DIA") || kelasRaw.includes("SAP")) kelas = "DIASAP3C"; else if (kelasRaw.includes("DCD") && kelasRaw.includes("2A")) kelas = "DCD2A"; rows.push({ id: String(row[colMap.id] || "").trim().toUpperCase(), name: String(row[colMap.name] || "").toUpperCase(), nickname: String(row[colMap.nickname] || "").replace(/\(.*?\)/g,"").trim(), phone: cleanPhone(row[colMap.phone]), mentor: String(row[colMap.mentor] || "").trim(), mentorPhone: "", guardian: String(row[colMap.guardian] || "").trim(), guardianPhone: cleanPhone(row[colMap.guardianPhone]), resident: String(row[colMap.resident] || "").toLowerCase().includes("non") ? "non-resident" : "resident", kelas: kelas, photoFileId: extractDriveId(row[colMap.photo] || "") }); } return output({ success: true, students: rows }); } function doPost(e){ var data = JSON.parse(e.postData.contents); if(data.action === "mark") return markSingle(data); if(data.action === "bulk") return saveBulk(data); return output({success:false}); } function markSingle(d){ var sh = getAttSheet(); sh.appendRow([new Date(), d.class, d.id, d.name, "HADIR", new Date().toLocaleTimeString(), d.device]); return output({success:true}); } function saveBulk(d){ var sh = getAttSheet(); d.records.forEach(r=>{ sh.appendRow([r.date, r.class, r.id, r.name, r.status, r.time, r.device]); }); return output({success:true}); } function getAttSheet(){ var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName("REKOD KEHADIRAN"); if(!sh){ sh = ss.insertSheet("REKOD KEHADIRAN"); sh.appendRow(["TARIKH","KEHADIRAN KELAS","NO ID","NAMA","STATUS","MASA","DEVICE"]); } return sh; } function getPhoto(studentId, fileId) { if (!fileId) return output({ success:false }); try { var file = DriveApp.getFileById(fileId); var blob = file.getBlob(); var b64 = "data:image/jpeg;base64," + Utilities.base64Encode(blob.getBytes()); return output({ success:true, id:studentId, photo:b64 }); } catch(err) { return output({ success:false }); } } function output(obj){ return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function extractDriveId(url) { if (!url) return ""; var m = String(url).match(/[-\w]{25,}/); return m ? m[0] : ""; } function normalize(str){ return String(str).toLowerCase().replace(/\s+/g,"").replace(/[^\w]/g,""); } function cleanPhone(p){ if (!p) return ""; p = String(p).replace(/\D/g,""); if (p.startsWith("0")) p = "6" + p; return p; }
